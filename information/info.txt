Class NetworkMonitorAgent(BaseAgent):

__init__:
  - Call super().__init__("network_monitor", KAFKA_TOPICS["network_monitor"])
  - Import NETWORK_INTERFACE from config.config
  - self.port_scan_tracker  = defaultdict(list)
    # {source_ip: [timestamps of each unique port contact]}
  - self.port_targets       = defaultdict(set)
    # {source_ip: set of unique destination ports seen}
  - self.syn_counter        = defaultdict(int)
    # {source_ip: SYN count in current second}
  - self.transfer_tracker   = defaultdict(int)
    # {(src_ip, dst_ip): bytes transferred}
  - self.lock               = threading.Lock()
  - self.reports_queue      = queue.Queue()

observe(self):
  - Use scapy AsyncSniffer on NETWORK_INTERFACE
  - Callback: self._packet_callback(pkt)
  - Run sniffer for 10 seconds
  - Return all queued reports: drain self.reports_queue into a list

analyze(self, raw_data):
  - raw_data is already a list of ThreatReport dicts from the queue
  - Return raw_data unchanged (detection happens in _packet_callback)

_packet_callback(self, pkt):
  Implements all 4 detections. For each detection that triggers,
  call self._emit(threat_report_dict) which puts it in reports_queue.

  DETECTION 1 — PORT SCAN:
    If pkt has IP and TCP layers:
      src = pkt[IP].src
      dst_port = pkt[TCP].dport
      with self.lock:
        self.port_targets[src].add(dst_port)
        self.port_scan_tracker[src].append(time.time())
        cutoff = time.time() - THRESHOLDS["port_scan_window_seconds"]
        self.port_scan_tracker[src] = [
            t for t in self.port_scan_tracker[src] if t > cutoff
        ]
        if len(self.port_targets[src]) > THRESHOLDS["port_scan_unique_ports"]:
          emit HIGH ThreatReport:
            threat_type = "port_scan"
            source_ip = src
            raw_evidence = f"Contacted {len(self.port_targets[src])} unique
                            ports in {THRESHOLDS['port_scan_window_seconds']}s"
            recommended_action = "Block source IP immediately, run
                                  nmap -sV {src} to fingerprint attacker"
          self.port_targets[src].clear()

  DETECTION 2 — SYN FLOOD:
    If pkt has TCP layer and pkt[TCP].flags == 0x02:
      with self.lock:
        self.syn_counter[pkt[IP].src] += 1
        if self.syn_counter[pkt[IP].src] > THRESHOLDS["syn_flood_per_second"]:
          emit HIGH ThreatReport:
            threat_type = "syn_flood"
            raw_evidence = f"SYN flood: {self.syn_counter[src]} SYN/s from {src}"
            recommended_action = "Rate-limit or null-route source IP"
          self.syn_counter[pkt[IP].src] = 0

  DETECTION 3 — DNS TUNNELING:
    If pkt has DNS layer and pkt.haslayer(DNSQR):
      qname = pkt[DNSQR].qname.decode(errors="ignore")
      payload_len = len(bytes(pkt[DNS]))
      labels = qname.split(".")
      long_label = any(len(l) > THRESHOLDS["dns_label_length"] for l in labels)
      if payload_len > THRESHOLDS["dns_payload_bytes"] or long_label:
        emit MEDIUM ThreatReport:
          threat_type = "dns_tunneling"
          source_ip = pkt[IP].src
          raw_evidence = f"DNS query: {qname[:80]} | payload={payload_len}b"
          recommended_action = "Inspect DNS traffic, block if confirmed C2"

  DETECTION 4 — LARGE OUTBOUND TRANSFER:
    If pkt has IP and TCP and Raw layers:
      key = (pkt[IP].src, pkt[IP].dst)
      with self.lock:
        self.transfer_tracker[key] += len(pkt[Raw].load)
        threshold_bytes = THRESHOLDS["large_transfer_mb"] * 1024 * 1024
        if self.transfer_tracker[key] > threshold_bytes:
          emit HIGH ThreatReport:
            threat_type = "data_exfiltration"
            source_ip = pkt[IP].src, destination_ip = pkt[IP].dst
            raw_evidence = f"Transfer of
              {self.transfer_tracker[key]//1024//1024}MB to {pkt[IP].dst}"
            recommended_action = "Isolate host, capture traffic with tcpdump"
          self.transfer_tracker[key] = 0

process_alert(self, threat_report: dict):
  source_ip = threat_report.get("source_ip", "0.0.0.0")
  with self.lock:
    self.correlation_window[source_ip].append(threat_report)
    // Remove events older than THRESHOLDS["correlation_window_seconds"]
    cutoff = time.time() - THRESHOLDS["correlation_window_seconds"]
    self.correlation_window[source_ip] = deque(
        [e for e in self.correlation_window[source_ip]
         if e["timestamp"] > cutoff],
        maxlen=100
    )
  result = self.correlate_and_score(source_ip)
  self.decide_response(result)

correlate_and_score(self, source_ip: str) -> dict:
  with self.lock:
    events = list(self.correlation_window[source_ip])

  if not events:
    return {"source_ip": source_ip, "compound_score": 0.0,
            "alert_count": 0, "agents_involved": [],
            "kill_chain": {}, "recommended_severity": "LOW"}

  // Base score
  base_score = sum(
      SEVERITY_SCORES.get(e["severity"], 1) for e in events
  )

  // Cross-agent multiplier
  agents_involved = list(set(e["agent_id"] for e in events))
  cross_agent_mult = 1.5 if len(agents_involved) > 1 else 1.0

  // Kill chain multiplier
  kc = analyze_kill_chain_progression(events)
  kc_mult = 1 + kc["progression_score"] * 0.2

  compound_score = base_score * cross_agent_mult * kc_mult

  // Derive severity
  if compound_score >= THRESHOLDS["compound_score_critical"]:
    recommended_severity = "CRITICAL"
  elif compound_score >= THRESHOLDS["compound_score_high"]:
    recommended_severity = "HIGH"
  elif compound_score >= THRESHOLDS["compound_score_medium"]:
    recommended_severity = "MEDIUM"
  else:
    recommended_severity = "LOW"

  return {
      "source_ip":            source_ip,
      "compound_score":       round(compound_score, 2),
      "alert_count":          len(events),
      "agents_involved":      agents_involved,
      "kill_chain":           kc,
      "recommended_severity": recommended_severity,
      "mitre_techniques":     list(set(
          e["mitre_technique_id"] for e in events
          if e.get("mitre_technique_id")
      ))
  }

decide_response(self, result: dict):
  sev = result["recommended_severity"]
  score = result["compound_score"]
  ip = result["source_ip"]

  if sev == "CRITICAL" or score >= THRESHOLDS["compound_score_critical"]:
    self.dispatcher.block_ip(ip)
    self.dispatcher.isolate_host(ip)
    self.dispatcher.send_slack_alert(
        self._format_alert(result), "CRITICAL")
    self.log_incident(result, "AUTO_BLOCKED_AND_ISOLATED")

  elif sev == "HIGH" or score >= THRESHOLDS["compound_score_high"]:
    self.dispatcher.block_ip(ip)
    self.dispatcher.send_slack_alert(self._format_alert(result), "HIGH")
    self.log_incident(result, "AUTO_BLOCKED")

  elif sev == "MEDIUM" or score >= THRESHOLDS["compound_score_medium"]:
    self.dispatcher.send_slack_alert(self._format_alert(result), "MEDIUM")
    self.log_incident(result, "PENDING_REVIEW")

  else:
    self.log_incident(result, "LOGGED")

log_incident(self, result: dict, action_taken: str):
  doc = {
      **result,
      "action_taken": action_taken,
      "incident_id":  str(uuid.uuid4()),
      "timestamp":    int(time.time())
  }
  try:
    self.es.index(index=ES_INDICES["incidents"], document=doc)
  except Exception as e:
    self.logger.error(f"ES write failed: {e}")
  // Print colored summary to stdout
  color = {"CRITICAL":"\033[41m","HIGH":"\033[31m",
           "MEDIUM":"\033[33m","LOW":"\033[36m"}
  c = color.get(result["recommended_severity"], "")
  reset = "\033[0m"
  print(f"{c}[INCIDENT] {result['recommended_severity']} | "
        f"IP: {result['source_ip']} | Score: {result['compound_score']} | "
        f"Action: {action_taken} | "
        f"Kill chain: {result['kill_chain'].get('narrative','')}{reset}")

_format_alert(self, result: dict) -> str:
  kc = result.get("kill_chain", {})
  return (
      f"*{result['recommended_severity']} THREAT DETECTED*\n"
      f"Source IP: `{result['source_ip']}`\n"
      f"Compound Score: {result['compound_score']}\n"
      f"Agents involved: {', '.join(result['agents_involved'])}\n"
      f"MITRE techniques: {', '.join(result.get('mitre_techniques',[]))}\n"
      f"Kill chain: {kc.get('narrative','Unknown')}\n"
      f"APT pattern: {'YES — ESCALATE NOW' if kc.get('is_apt_pattern') else 'No'}"
  )

run(self):
  self.logger.info("Orchestrator started, consuming from all agent topics")
  while self.running:
    msg = self.consumer.poll(timeout=1.0)
    if msg is None: continue
    if msg.error():
      self.logger.error(f"Kafka error: {msg.error()}")
      continue
    try:
      threat_report = json.loads(msg.value().decode("utf-8"))
      self.process_alert(threat_report)
    except json.JSONDecodeError as e:
      self.logger.error(f"Bad JSON from Kafka: {e}")
    except Exception as e:
      self.logger.error(f"Processing error: {e}")

Handle SIGINT/SIGTERM: set self.running=False,
  self.consumer.close(), log "Orchestrator stopped"
